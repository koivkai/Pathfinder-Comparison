package datastructures;

public class CordinatePair {

    private Cordinate targetCordinate;
    private Cordinate parentCordinate;

    public CordinatePair(Cordinate targetCordinate, Cordinate prevCordinate) {
        this.targetCordinate = targetCordinate;
        this.parentCordinate = prevCordinate;
    }

    public Cordinate getTargetCordinate() {
        return targetCordinate;
    }

    public void setTargetCordinate(Cordinate targetCordinate) {
        this.targetCordinate = targetCordinate;
    }

    public Cordinate getPrevCordinate() {
        return parentCordinate;
    }

    public void setPrevCordinate(Cordinate prevCordinate) {
        this.parentCordinate = prevCordinate;
    }

    
    
}
